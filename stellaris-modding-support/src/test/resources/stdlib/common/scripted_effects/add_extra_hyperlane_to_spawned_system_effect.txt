add_extra_hyperlane_to_spawned_system_effect = {
	hidden_effect = {
		last_created_system = {
			if = {
				limit = {
					any_system = {
						distance = {
							source = prev
							type = euclidean
							min_distance >= 1
							max_distance <= 10
						}
						NOT = { has_hyperlane_to = prev }
					}
				}
				random_system = {

							max_distance <= 10

					}
					add_hyperlane = {
						from = this
						to = prev
					}
				}
			}
			else_if = {
				limit = {
					any_system = {
