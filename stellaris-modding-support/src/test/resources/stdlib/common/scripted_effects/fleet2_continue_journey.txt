fleet2_continue_journey = {
	# Untag capital planet
	log = "triggered fleet2_continue_journey"
	from = {
		remove_planet_flag = fleet2_capital_destination
		log = "fleet 2 untagged planet"
	}
	# Move on to next capital
	if = {
		limit = {
			any_system = {
				NOT = { is_same_value = solar_system }
				any_system_planet = { has_planet_flag = fleet2_capital_destination }
			}
		}
		solar_system = {
			closest_system = {
				min_steps = 1
				limit = {
					any_system_planet = { has_planet_flag = fleet2_capital_destination }
				}
				random_system_planet = {
					limit = { has_planet_flag = fleet2_capital_destination }
					save_event_target_as = fleet2_next_capital_destination
					log = "fleet 2 set next destination"
				}
			}
		}
		auto_move_to_planet = {
			target = event_target:fleet2_next_capital_destination
			clear_auto_move_on_arrival = no
			arrival_effect = fleet2_continue_journey
		}
		log = "fleet 2 engaged auto_move"
	}
	# Or return home and reset in 10 years
	else_if = {
		limit = { exists = event_target:caravaneer_home_star }
		auto_move_to_planet = {
			target = event_target:caravaneer_home_star
			clear_auto_move_on_arrival = no
			arrival_effect = fleet2_reset_journey
		}
		log = "fleet 2 returning home"
	}
	else = {
		random_planet = { save_event_target_as = temporary_fleet2_destination }
		auto_move_to_planet = {
			target = event_target:caravaneer_home_star
			clear_auto_move_on_arrival = no
			arrival_effect = fleet2_continue_journey
		}
		log = "fleet 2 taking a detour, home is dead/not yet reestablished"
	}
}
