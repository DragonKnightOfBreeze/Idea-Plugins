building_basic_income_check = {
	planet = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			AND = {
				free_jobs < 3
				owner = {
					OR = {
						has_monthly_income = {
							resource = energy
							value > -5
						}
						has_resource = { type = energy amount > 1500 }
					}

					OR = {
						has_monthly_income = {
							resource = minerals
							value > 0
						}
						has_resource = { type = minerals amount > 1500 }
					}
				}
			}
		}
	}
}
