jobs_work_minerals_goods = {
	exists = owner
	owner = {
		is_ai = yes
		has_monthly_income = {
			resource = minerals
			value < 50
		}
		OR = {
			AND = {
				has_resource = {
					type = minerals
					amount < 5000
				}
				has_resource = {
					type = consumer_goods
					amount > 1500
				}
			}
			AND = {
				has_resource = {
					type = minerals
					amount < 3000
				}
				has_resource = {
					type = consumer_goods
					amount > 750
				}
			}
			AND = {
				has_resource = {
					type = minerals
					amount < 501
				}
				has_resource = {
					type = consumer_goods
					amount > 500
				}
			}
		}
	}
}#If true makes the AI really value mining jobs and stops organics from working factory jobs.

buildings_has_motes = {
	exists = owner
	OR = {
		owner = {
			has_monthly_income = {
				resource = volatile_motes
				value >= 3
			}
		}
		AND = {
			free_building_slots > 1
			owner = {
				has_monthly_income = {
					resource = volatile_motes
					value > 0
				}
			}
		}
	}
}#Used in the buildings file to determine if the country has the needed Strategic Resource.
