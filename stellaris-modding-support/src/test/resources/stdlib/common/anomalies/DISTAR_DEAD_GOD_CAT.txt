DISTAR_DEAD_GOD_CAT = {
	desc = "DISTAR_DEAD_GOD_CAT_DESC"
	picture = "GFX_evt_ship_in_orbit"
	level = 2
	
	spawn_chance = {
		modifier = {
			add = 1000
			has_distar = yes
			is_moon = yes
			solar_system = {
				any_system_planet = {
					exists = owner
					owner = {
						is_primitive = yes
					}
					has_moon = yes
					any_moon = {
						is_same_value = root
						habitable_planet = no
					}
				}
			}
		}
	}

	on_spawn = {
		solar_system = {
			random_system_planet = {
				limit = {
					exists = owner
					owner = { is_primitive = yes }
					has_moon = yes
					any_moon = {
						is_same_value = root
						habitable_planet = no
					}
				}
				add_modifier = { modifier = dead_god_planet_hidden days = -1 } 
			}
		}
	}
	
	on_success = distar.80
}
