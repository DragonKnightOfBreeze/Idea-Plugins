hegemony_federation_passive = {
	members_modifier = {
		description = hegemony_federation_passive_effect
	}
	on_activate = {
		set_diplomacy_action_setting = {
			action = action_ask_to_leave_federation
			settings = {
				acceptance_type = leader
			}
		}
	}
	on_deactivate = {
		set_diplomacy_action_setting = {
			action = action_ask_to_leave_federation 
			settings = {
				acceptance_type = default
			}
		}
	}
}
