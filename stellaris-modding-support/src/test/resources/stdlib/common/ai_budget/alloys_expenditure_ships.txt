alloys_expenditure_ships = {
	resource = alloys
	type = expenditure
	category = ships

	potential = {
		always = yes
	}

	fraction = {
		weight = 0.6

		# spend more alloys building ships while at war
		modifier = {
			factor = 2
			is_at_war = yes
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}
	}
}
